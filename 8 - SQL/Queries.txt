-- Q1 - Retrieve all customers who have placed an order in the last 30 days.
-- Retrieves unique customers by joining the tables and filtering with a date interval.
SELECT DISTINCT c.id, c.name, c.email 
FROM customers c 
JOIN orders o ON c.id = o.customer_id 
WHERE o.order_date >= CURRENT_DATE - INTERVAL 30 DAY;


-- Q2 - Get the total amount of all orders placed by each customer.
-- Calculates total spending per customer using the SUM() aggregate function and groups the results.
SELECT c.name, SUM(o.total_amount) AS total_spent
FROM customers c
JOIN orders o ON c.id = o.customer_id
GROUP BY c.id, c.name
ORDER BY total_spent DESC;


-- Q3 - Update the price of Product C to 45.00.
-- Modifies the price of a specific item (Wireless Mouse) using a WHERE clause for safety.
UPDATE products

-- Q5 - Retrieve the top 3 products with the highest price.
-- Sorts the products from most to least expensive and restricts the output to the top 3.
SELECT id, name, price
FROM products
ORDER BY price DESC
LIMIT 3;


-- Created Table for Product, Order and Customer Relation
-- Acts as a bridge (junction table) to link specific products to specific orders.
CREATE TABLE order_items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT,
    product_id INT,
    quantity INT,
    FOREIGN KEY (order_id) REFERENCES orders(id),
    FOREIGN KEY (product_id) REFERENCES products(id)
);


-- Q6 - Get the names of customers who have ordered Product A.
-- Uses multiple JOINs across four tables to find users who bought a specific item.
SELECT DISTINCT c.name
FROM customers c
JOIN orders o ON c.id = o.customer_id
JOIN order_items oi ON o.id = oi.order_id
JOIN products p ON oi.product_id = p.id
WHERE p.name = '27-inch 4K Monitor';


-- Q7 - Join the orders and customers tables to retrieve the customer's name and order date for each order.
-- Performs a standard INNER JOIN to match customer records with their corresponding orders.
SELECT c.name, o.order_date
FROM customers c
JOIN orders o ON c.id = o.customer_id;


-- Q8 - Retrieve the orders with a total amount greater than 150.00.
-- Filters the dataset using a mathematical comparison operator.
SELECT *
FROM orders
WHERE total_amount > 150.00;


-- Q9 - Normalize the database by creating a separate table for order items and updating the orders table to reference the order_items table.
-- Resolves the many-to-many relationship by establishing proper foreign key references.
CREATE TABLE order_items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT,
    product_id INT,
    quantity INT,
    FOREIGN KEY (order_id) REFERENCES orders(id),
    FOREIGN KEY (product_id) REFERENCES products(id)
);


-- Q10 - Retrieve the average total of all orders
-- Calculates the mathematical average across the entire total_amount column.
SELECT AVG(total_amount) AS average_order_value
FROM orders;
